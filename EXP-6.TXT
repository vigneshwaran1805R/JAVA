File 1: Student.java (inside package)
package studentpack;
public class Student {
private String name;
private int regNo;
// Setter methods
public void setName(String n) {
name = n;
}
public void setRegNo(int r) {
regNo = r;
}
// Getter methods
public String getName() {
return name;
}
public int getRegNo() {
return regNo;
}
}

import studentpack.Student;
public class PackageEncapsulationDemo {
public static void main(String[] args) {
Student s = new Student();

s.setRegNo(101);

}
}
